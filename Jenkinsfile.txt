pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                // Крок 1: Клонування репозиторію з GitHub
                git url: 'https://github.com/bohdankaaa6-web/jenkins-ci-project.git', branch: 'main'
            }
        }
        
        stage('Build') {
            steps {
                // Використовуємо msBuildFile та msBuildName (вимоги плагіна)
                msbuild msBuildFile: 'test_repos.sln', 
                        configuration: 'Debug', 
                        msBuildName: 'VS2022_Community' 
            }
        }

        stage('Test & Report') {
            steps {
                // Запуск .exe, який генерує test_report.xml
                // Зверніть увагу на шлях x64\Debug
                bat 'x64\\Debug\\test_repos.exe --gtest_output="xml:x64/Debug/test_report.xml"'
            }
        }
    }

    post {
        always {
            // Правильний синтаксис xunit
            // Публікація результатів тестів Google Test
            xunit([GoogleTest(deleteOutputFiles: true, failIfNotNew: false, pattern: 'x64/Debug/test_report.xml', skipNoTestFiles: false, stopProcessingIfError: true)])
        }
    }
}
